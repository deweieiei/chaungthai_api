┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: users  — ข้อมูลผู้ใช้งานทุกประเภท
└─────────────────────────────────────────────────────────────────────────────
  id                   BIGINT UNSIGNED  PK AUTO_INCREMENT
                       รหัสผู้ใช้ไม่ซ้ำกัน ใช้เป็น Foreign Key ในตารางอื่นทั่วระบบ

  email                VARCHAR(255)  UNIQUE  NOT NULL
                       อีเมลที่ใช้สมัครหรือ login ต้องไม่ซ้ำกัน

  phone                VARCHAR(20)  UNIQUE
                       เบอร์โทรศัพท์ (อาจเป็น NULL ถ้าสมัครด้วยอีเมลอย่างเดียว)
                       ต้องไม่ซ้ำกัน เพราะ 1 เบอร์ = 1 บัญชีเท่านั้น

  password_hash        VARCHAR(255)  NOT NULL
                       รหัสผ่านที่ผ่านการเข้ารหัสด้วย bcrypt หรือ argon2
                       ห้ามเก็บรหัสผ่านดิบ (plain text) เด็ดขาด

  first_name           VARCHAR(100)  NOT NULL   ชื่อจริง
  last_name            VARCHAR(100)  NOT NULL   นามสกุล

  role                 ENUM('customer','worker','admin')  DEFAULT 'customer'
                       บทบาท: customer=ผู้ว่าจ้าง  worker=ช่าง  admin=ผู้ดูแล
                       ใช้ filter สิทธิ์การเข้าถึง API ทั่วระบบ

  email_verified_at    DATETIME  (NULL ได้)
                       วันเวลาที่ยืนยันอีเมลสำเร็จ  NULL = ยังไม่ยืนยัน
                       ใช้เช็ค: WHERE email_verified_at IS NOT NULL

  phone_verified_at    DATETIME  (NULL ได้)
                       วันเวลาที่ยืนยันเบอร์โทรสำเร็จ

  is_active            TINYINT(1)  DEFAULT 1
                       1=บัญชีใช้งานได้  0=ถูกระงับ
                       ทุก API endpoint ควร filter WHERE is_active=1

  newsletter_opt_in    TINYINT(1)  DEFAULT 0
                       1=ยินยอมรับอีเมลโปรโมชัน  0=ไม่ต้องการ

  last_login_at        DATETIME   วันเวลา login ล่าสุด (อัพเดททุกครั้งที่ login)
  created_at           DATETIME   วันสร้างบัญชี
  updated_at           DATETIME   วันแก้ไขล่าสุด


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: worker_profiles  — โปรไฟล์เพิ่มเติมของ worker (1-to-1 กับ users)
└─────────────────────────────────────────────────────────────────────────────
  user_id              BIGINT UNSIGNED  PK + FK → users.id
                       ใช้ user_id เป็น PK ด้วย แสดงว่า 1 worker = 1 profile เท่านั้น

  bio                  TEXT   คำแนะนำตัว/ประวัติย่อของช่าง แสดงในโปรไฟล์สาธารณะ

  skills               JSON   (legacy) รายการทักษะ เช่น ["ช่างไฟ","ช่างประปา"]
                       NOTE: ตาราง worker_skills ใหม่จัดการทักษะแบบ relational แทน
                       ฟิลด์นี้เก็บไว้สำหรับ backward compat หรือ quick display

  service_areas        JSON   (legacy) พื้นที่บริการ เช่น ["กรุงเทพฯ","นนทบุรี"]
                       NOTE: ตาราง worker_service_areas ใหม่จัดการแบบ relational แทน

  avatar_url           VARCHAR(500)  URL รูปโปรไฟล์ช่าง
  rating_avg           DECIMAL(3,2)  DEFAULT 0.00   คะแนนเฉลี่ย 0.00–5.00
  rating_count         INT UNSIGNED  DEFAULT 0       จำนวนรีวิวทั้งหมด

  tickets_balance      INT UNSIGNED  DEFAULT 5
                       ยอดตั๋วคงเหลือปัจจุบัน ใช้ตรวจสอบก่อนอนุญาตให้สมัครงาน
                       อัพเดทพร้อมกันกับการเขียน ticket_transactions เสมอ

  is_verified          TINYINT(1)  DEFAULT 0
                       0=ยังไม่ยืนยัน  1=ผ่าน KYC แล้ว (ได้รับ Verified Badge)
                       ช่างที่ is_verified=0 รับงานได้เฉพาะ job_level 1-2

  verified_at          DATETIME   วันที่ admin อนุมัติ KYC
  created_at / updated_at   DATETIME


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: otp_codes  — รหัส OTP สำหรับยืนยันตัวตน
└─────────────────────────────────────────────────────────────────────────────
  id           BIGINT UNSIGNED  PK
  user_id      BIGINT UNSIGNED  FK → users.id  (NULL ได้ กรณีสมัครใหม่ยังไม่มี account)
  target       VARCHAR(255)     ปลายทางที่ส่ง OTP (เบอร์โทรหรืออีเมล)
  code_hash    CHAR(64)         SHA-256 hash ของรหัส OTP 6 หลัก
  purpose      ENUM('verify_email','verify_phone','reset_password')
  attempts     TINYINT UNSIGNED  DEFAULT 0   จำนวนครั้งที่กรอกผิด (ล็อกเมื่อ >= 5)
  used_at      DATETIME   วันที่ใช้ OTP สำเร็จ (NULL = ยังไม่ได้ใช้)
  expires_at   DATETIME   วันหมดอายุ (ปกติ 5-10 นาที)
  created_at   DATETIME

  การ query ที่ถูกต้อง:
    SELECT * FROM otp_codes
    WHERE target=? AND purpose=? AND used_at IS NULL AND expires_at > NOW()
    ORDER BY created_at DESC LIMIT 1


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: refresh_tokens  — JWT Refresh Token
└─────────────────────────────────────────────────────────────────────────────
  id           BIGINT UNSIGNED  PK
  user_id      BIGINT UNSIGNED  FK → users.id  NOT NULL
  token_hash   CHAR(64)  UNIQUE   SHA-256 ของ token จริง (ไม่เก็บ token ดิบ)
  user_agent   VARCHAR(255)       ระบุอุปกรณ์/เบราว์เซอร์
  ip           VARCHAR(45)        IPv4 หรือ IPv6
  expires_at   DATETIME           วันหมดอายุ token
  revoked_at   DATETIME   วันที่ยกเลิก token (NULL = ยังใช้งานได้)
  created_at   DATETIME

  token ที่ valid: revoked_at IS NULL AND expires_at > NOW()

┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: skill_categories  — หมวดหมู่ทักษะ
└─────────────────────────────────────────────────────────────────────────────
  id          INT UNSIGNED  PK AUTO_INCREMENT
  name_th     VARCHAR(100)  ชื่อภาษาไทย  เช่น "คอมพิวเตอร์และไอที"
  name_en     VARCHAR(100)  ชื่อภาษาอังกฤษ เช่น "Computer & IT"
  icon_url    VARCHAR(500)  URL ไอคอน (ไฟล์รูปหรือ icon library)
  is_active   TINYINT(1)  DEFAULT 1   0=ซ่อนจากหน้าแอป
  sort_order  SMALLINT UNSIGNED  ลำดับแสดงผลในหน้า UI
  created_at  DATETIME

  ตัวอย่าง: คอมพิวเตอร์/ไอที, ช่างไฟฟ้า, ช่างประปา, ทำความสะอาด, ดูแลสวน ฯลฯ


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: skills  — ทักษะย่อย
└─────────────────────────────────────────────────────────────────────────────
  id           INT UNSIGNED  PK AUTO_INCREMENT
  category_id  INT UNSIGNED  FK → skill_categories.id
  name_th      VARCHAR(100)  ชื่อทักษะ เช่น "ติดตั้ง Windows"
  name_en      VARCHAR(100)
  description  VARCHAR(255)  คำอธิบายสั้น
  is_active    TINYINT(1)
  sort_order   SMALLINT UNSIGNED
  created_at   DATETIME

  ตัวอย่าง (category: คอมพิวเตอร์):
    - ติดตั้ง/ลงโปรแกรม Windows
    - ซ่อมฮาร์ดแวร์คอมพิวเตอร์
    - แก้ปัญหาไวรัส/มัลแวร์
    - ติดตั้งกล้องวงจรปิด


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: worker_skills  — ทักษะของช่างแต่ละคน + ราคา + ระดับ
└─────────────────────────────────────────────────────────────────────────────
  id           BIGINT UNSIGNED  PK AUTO_INCREMENT
  worker_id    BIGINT UNSIGNED  FK → worker_profiles.user_id
  skill_id     INT UNSIGNED     FK → skills.id
  UNIQUE KEY (worker_id, skill_id)  — ช่างคนนึงมีทักษะเดียวกันได้แค่ 1 แถว

  skill_level  TINYINT UNSIGNED  NOT NULL  DEFAULT 1
               ระดับความสามารถที่ช่างประเมินตัวเอง:
               1 = มือใหม่ (beginner)
               2 = พอใช้ (intermediate)
               3 = ดี (proficient)
               4 = เชี่ยวชาญ (advanced)
               5 = มืออาชีพ (expert/master)

  price_type   ENUM('fixed','per_hour','negotiable')  DEFAULT 'fixed'
               fixed      = ราคาคงที่ต่องาน (ดู price)
               per_hour   = คิดรายชั่วโมง (ดู price)
               negotiable = ต่อรองได้ (price อาจเป็น 0 หรือราคาเริ่มต้น)

  price        DECIMAL(10,2)  DEFAULT 0.00
               ราคาเป็นบาท (0 = ต่อรองได้, ไม่ได้แปลว่าฟรี)

  description  TEXT   รายละเอียดเพิ่มเติม เช่น "ลง Windows พร้อม Office เริ่ม 200"
  created_at / updated_at  DATETIME

  ตัวอย่างการ query หาช่างที่มีทักษะ skill_id=1 เรียงตาม ระดับและราคา:
    SELECT u.first_name, ws.skill_level, ws.price_type, ws.price
    FROM worker_skills ws
    JOIN users u ON u.id = ws.worker_id
    WHERE ws.skill_id = 1 AND ws.worker_id IN (
      SELECT user_id FROM worker_service_areas WHERE province_id = 1
    )
    ORDER BY ws.skill_level DESC, ws.price ASC

┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: worker_service_areas  — พื้นที่ที่ช่างรับงาน
└─────────────────────────────────────────────────────────────────────────────
  id              BIGINT UNSIGNED   PK AUTO_INCREMENT

  province_id     SMALLINT UNSIGNED FK → provinces.id   (บังคับ)

    (worker_id=9001, province_id=1, district_id=2)


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: worker_locations  — พิกัด GPS ของช่าง (real-time ในอนาคต)
└─────────────────────────────────────────────────────────────────────────────
  worker_id         BIGINT UNSIGNED  PK + FK → worker_profiles.user_id
                    (1 ช่าง = 1 แถว)

  latitude          DECIMAL(10,7)   ละติจูด เช่น 13.8196440
  longitude         DECIMAL(10,7)   ลองจิจูด เช่น 100.5581580

  service_radius_km SMALLINT UNSIGNED  DEFAULT 10
                    รัศมีรับงานจากตำแหน่งปัจจุบัน (กิโลเมตร)
                    ลูกค้าอยู่ใน radius นี้ถึงจะเห็นช่าง

  updated_at        DATETIME  อัพเดททุกครั้งที่ app เปิด (หรือเมื่อเปิดรับงาน)

  คำนวณระยะทาง (Haversine formula) ใน SQL:
    SELECT worker_id,
      (6371 * ACOS(COS(RADIANS(13.7563)) * COS(RADIANS(latitude))
       * COS(RADIANS(longitude) - RADIANS(100.5018))
       + SIN(RADIANS(13.7563)) * SIN(RADIANS(latitude)))) AS distance_km
    FROM worker_locations
    HAVING distance_km <= service_radius_km
    ORDER BY distance_km;

 กลุ่มที่ 4: Job Lifecycle (ใหม่) — วงจรชีวิตของงาน
══════════════════════════════════════════════════════════════════════════════

  ขั้นตอน (flow):
  [ลูกค้าโพสงาน] → jobs (status=open)
       ↓ ช่างเห็น → สมัคร
  [ช่างสมัคร] → job_applications (status=pending) + หักตั๋ว
       ↓ ลูกค้าเลือก
  [ลูกค้าเลือกช่าง] → job_applications (status=accepted)
                     → jobs (status=matched)
                     → job_matches (status=matched)
                     → chat_rooms สร้างขึ้น
       ↓ เริ่มงาน
  [งานเริ่ม] → job_matches (status=in_progress)
       ↓ งานเสร็จ
  [งานเสร็จ] → job_matches (status=completed)
             → ratings สร้างขึ้น (ลูกค้าให้ดาว)
             → worker_profiles.rating_avg/rating_count อัพเดท


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: jobs  — ประกาศงาน (โพสหาช่าง)
└─────────────────────────────────────────────────────────────────────────────
  id               BIGINT UNSIGNED  PK AUTO_INCREMENT
  customer_id      BIGINT UNSIGNED  FK → users.id   ผู้โพสงาน

  title            VARCHAR(255)  หัวข้องาน เช่น "ต้องการช่างลง Windows 11"
  description      TEXT          รายละเอียดงานเต็ม

  job_level        TINYINT UNSIGNED  DEFAULT 1
                   ระดับความซับซ้อนของงาน กำหนดจำนวนตั๋วที่ช่างต้องใช้:

                   2 = งานปานกลาง-ต่ำ (2 ตั๋ว)
                   3 = งานปานกลาง (3 ตั๋ว) เช่น ล้างแอร์, ลง Windows
                   4 = งานยาก (4 ตั๋ว) เช่น ซ่อมไฟฟ้า, ต่อท่อน้ำ
                   5 = งานซับซ้อน (5 ตั๋ว) เช่น งานก่อสร้าง, ต่อเติม

  budget_type      ENUM('fixed','open')  DEFAULT 'open'
                   fixed = ลูกค้ากำหนดราคาเอง (ดู budget_amount)
                   open  = ให้ช่างเสนอราคา (budget_amount = NULL)

  budget_amount    DECIMAL(10,2)  ราคาที่ลูกค้ายินดีจ่าย (NULL ถ้า open)

  province_id      SMALLINT UNSIGNED  FK → provinces.id
  district_id      SMALLINT UNSIGNED  FK → districts.id
  subdistrict_id   INT UNSIGNED       FK → subdistricts.id
  latitude         DECIMAL(10,7)      GPS ละติจูด (optional, ใช้ในอนาคต)
  longitude        DECIMAL(10,7)      GPS ลองจิจูด
  address_note     VARCHAR(255)       หมายเหตุที่อยู่ เช่น "ห้อง 5B อาคาร A"

  status           ENUM:
                   open        = รับสมัครช่างอยู่
                   matched     = ลูกค้าเลือกช่างแล้ว รอเริ่มงาน
                   in_progress = งานกำลังดำเนินการ

                   cancelled   = ลูกค้ายกเลิก

  view_count       INT UNSIGNED  DEFAULT 0    จำนวนครั้งถูกเปิดดู

  completed_at     DATETIME   วันที่งานเสร็จ
  cancelled_at     DATETIME
  cancel_reason    VARCHAR(255)
  created_at / updated_at  DATETIME

    WHERE participant_a=? OR participant_b=?

  reject_reason           VARCHAR(255)    เหตุผลที่ reject
  reviewed_by             BIGINT UNSIGNED user_id ของ admin ที่ตรวจสอบ
  reviewed_at             DATETIME
  submitted_at            DATETIME        NOT NULL  DEFAULT CURRENT_TIMESTAMP

  NOTE: ลบหลังจาก 1 ปี หลังปิดบัญชีหรือถูกแบน (ตาม Privacy Policy)


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: blacklist  — บัญชีถูกแบน ป้องกันสมัครซ้ำ (DB2)
└─────────────────────────────────────────────────────────────────────────────
  id               BIGINT UNSIGNED  PK AUTO_INCREMENT
  identifier_hash  CHAR(64)  SHA-256 hash ของเบอร์โทรหรืออีเมล
                   (ไม่เก็บข้อมูลดิบ เพราะ PDPA)
  identifier_type  ENUM('phone','email','device_id')
  user_id          BIGINT UNSIGNED  อ้างอิง (NULL ได้ ถ้าบัญชีถูกลบแล้ว)
  reason           VARCHAR(255)     เหตุผลที่แบน
  banned_by        BIGINT UNSIGNED  user_id ของ admin
  banned_at        DATETIME         NOT NULL  DEFAULT CURRENT_TIMESTAMP
  expires_at       DATETIME         NULL = แบนถาวร, มีค่า = แบนชั่วคราว

  การเช็ค: SELECT * FROM blacklist
           WHERE identifier_hash = SHA2(?, 256)
           AND (expires_at IS NULL OR expires_at > NOW())


┌─────────────────────────────────────────────────────────────────────────────
│ TABLE: sensitive_access_logs  — log การเข้าถึงข้อมูล sensitive (DB2)
└─────────────────────────────────────────────────────────────────────────────
  id             BIGINT UNSIGNED  PK AUTO_INCREMENT
  accessed_by    BIGINT UNSIGNED  user_id ของคนที่เข้าถึง (admin)
  resource_type  VARCHAR(64)      'kyc_submission' | 'blacklist' | 'id_card_file'
  resource_id    BIGINT UNSIGNED  ID ของ resource
  action         VARCHAR(64)      'view' | 'download' | 'approve' | 'reject' | 'delete'
  ip             VARCHAR(45)
  user_agent     VARCHAR(255)
  created_at     DATETIME  NOT NULL  DEFAULT CURRENT_TIMESTAMP

  บันทึกทุกครั้งที่มีการเข้าถึงข้อมูล sensitive (ข้อกำหนด PDPA)
  ห้ามลบ — เก็บตลอดกาล
